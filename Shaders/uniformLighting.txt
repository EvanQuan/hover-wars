// Structures for Different Light Sources
struct DirLight {
	vec3 vDirection;
	vec3 vAmbient;
	vec3 vDiffuse;
	vec3 vSpecular;
};

struct PointLight {
	float fLightPower;
	vec3 vPosition;
	vec3 vColor;
};

struct SpotLight {
	float fCosPhi;
	float fCosCutoff;
	vec3 vPosition;
	vec3 vDirection;
	vec3 vColor;
};

layout (std140, binding = 2) uniform Lights
{
	int numPointLights;
	int numSpotLights;
	bool usingDirectionalLight;
	DirLight pDirectionalLight;
	PointLight pPointLights[12];
	SpotLight pSpotLights[12];
};

// The Threshold to determine if the fragment is sufficiently bright enough to bloom
const vec3 THRESHOLD = vec3(0.2126, 0.7152, 0.0722);

// Shadow Map Uniforms
uniform sampler2D DirectionalLightShadow;
