// Structures for Different Light Sources
struct DirLight {
	vec3 vDirection;
	vec3 vAmbient;
	vec3 vDiffuse;
	vec3 vSpecular;
};

struct PointLight {
	float fLightPower;
	vec3 vPosition;
	vec3 vColor;
};

struct SpotLight {
	float fCosPhi;
	float fCosCutoff;
	vec3 vPosition;
	vec3 vDirection;
	vec3 vColor;
};

layout (std140, binding = 2) uniform Lights
{
	int numPointLights;
	int numSpotLights;
	bool usingDirectionalLight;
	DirLight pDirectionalLight;
	PointLight pPointLights[4];
	SpotLight pSpotLights[4];
};

// Shadow Map Uniforms
uniform sampler2D DirectionalLightShadow;
uniform sampler2D SpotLightShadows[4];
